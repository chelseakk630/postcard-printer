    <canvas id="postcardCanvas" width="1115" height="768"></canvas>

  </div>


  <script>

    function generatePostcard() {

      const name = document.getElementById("nameInput").value.trim();

      if (!name) {

        alert("Please enter a name.");

        return;

      }


      const canvas = document.getElementById("postcardCanvas");

      const ctx = canvas.getContext("2d");


      const image = new Image();

      image.onload = () => {

        ctx.clearRect(0, 0, canvas.width, canvas.height);

        ctx.drawImage(image, 0, 0, canvas.width, canvas.height);


        // Set font and color

        ctx.font = "28px cursive";

        ctx.fillStyle = "#1A3E34";

        ctx.fillText("Dear " + name + ",", 90, 230); // Adjust text position if needed


        // Open image in new window and trigger print

        const win = window.open();

        win.document.write('<img src="' + canvas.toDataURL() + '" onload="window.print();window.close();" />');

      };


      image.src = "5e06c33d5bf03ed9902f76597f9809a5.jpg"; // Make sure the image is in the same folder

    }

  </script>
